function addColumn() {
  while (true) {
    var x = prompt("Please enter unique column name");
    if (x != null) {
      for (columnNum in table.getColumnDefinitions()){
          if (table.getColumnDefinitions()[columnNum].title == x) {
             x = "";
             break;
          }
      }
    }
    if (x == null || x == "") continue;
    table.addColumn({title:x, field:x}, true, x);
    break;
  }
}

var tabledata = [
    {id:1, name:"abc", age:"12"},
    {id:1, name:"def", age:"14"},
 ];

var table = new Tabulator("#data-table", {
    height:205,
    data:tabledata,
    layout:"fitColumns",
    autoColumns:true,
movableColumns:true,
history:true,
addRowPos:"top",
paginationSize:7,
    paginationCounter:"rows",
 movableRows: true,
    movableRowsConnectedElements: "#drop-area",
movableColumns: true
});

table.on("movableRowsElementDrop", function(e, element, row){
        var div = document.createElement("div");
        div.textContent = row.getData().name;
        element.appendChild(div);
});

</script>
