        <script>
            function addFile() {
                while (true) {
                    var x = prompt("Please enter new file name");
                    if (x == null) break;
                    if (x == "") continue;
                    executePOST("op=newfile&" + "name=" + x,
                        function(...args) {
                            window.location.href = '/';
                        },
                        x,
                    );
                    break;
                }
            }
        </script>
